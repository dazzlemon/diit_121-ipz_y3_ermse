\begin{alignat*}{2}%
x_n = x_{10} = {10.000} \\%
x_{arif}
        = \frac {x_1 + x_n} 2
        = \frac {1.000 + 10.000} 2
        = 5.500 \\%
x_{geom}
                    = \sqrt{x_1 * x_n}
                    = \sqrt{1.000 * 10.000}
                    = 3.162 \\%
x_{garm}
                    = \frac {2 * x_1 * x_n} {x_1 + x_n}
                    = \frac 
                        {2 * 1.000 * 10.000}
                        {1.000 + 10.000}
                    = 1.818%
\end{alignat*}
